securecookies

Customizable middleware for adding automatic cookie encryption and decryption to Starlette applications.

How it works?

For any incoming cookies:

  1. Requests sent from the client's browser to your application are intercepted by SecureCookiesMiddleware.
  2. All Cookie headers are parsed and filter. Only cookies in the included_cookies and excluded_cookies parameters are parsed. All cookies are included by default.
  3. The cookies are decrypted. If cookie cannot be decrypted, or is otherwise invalid, it is discarded by default (discard_invalid=True).
  4. Any included and valid encrypted cookies in the ASGI request scope are replaced by the decrypted ones.
  5. The request scope is passed to any future middleware, and eventually your application. Cookies can be read normally anywhere downstream.

For any outgoing cookies:

  1. Your application sets cookies with response.set_cookie as usual.
  2. All responses returned by your application are intercepted by SecureCookiesMiddleware.
  3. Cookies in the included_cookies and excluded_cookies parameters are re-encrypted, and their attributes (like "SameSite" and "HttpOnly") are overridden by the parameters set in SecureCookiesMiddleware.
  4. The cookies in the response are replaced by the re-encrypted cookies, and the response is propagated to Starlette to return to the client's browser.

 19    """
 20    Provides automatic secure cookie support through symmetric Fernet encryption. As per
 21    the `cryptography` docs, security is achieved through:
 22
 23    - 128-bit key AES in CBC mode using PKCS7 padding for encryption
 24    - HMAC using SHA256 for authentication.
 25    """
 26
 27    secrets: List[str]
 28    """
 29    A list of secrets to use when encrypting and decrypting secure cookies. Supplying
 30    multiple secrets enables the use of key rotation, whereby the first key is used for
 31    all encryption, including to re-encrypt cookies previously encrypted with any of
 32    the other list entries.
 33    """

 35    def __init__(
 36        self,
 37        app: ASGIApp,
 38        secrets: List[str],
 39        discard_invalid: bool = True,
 40        cookie_path: Optional[str] = None,
 41        cookie_domain: Optional[str] = None,
 42        cookie_secure: Optional[bool] = None,
 43        cookie_httponly: Optional[bool] = None,
 44        cookie_samesite: Optional[Literal["lax", "strict", "none"]] = None,
 45        excluded_cookies: Optional[List[str]] = None,
 46        included_cookies: Optional[List[str]] = None,
 47    ) -> None:
 48        super().__init__(app)
 49        self.mfernet = MultiFernet(Fernet(s) for s in secrets)
 50        self.discard_invalid = discard_invalid
 51        """ Discard any invalid / unencrypted cookies present in the request."""
 52        self.cookie_path = cookie_path
 53        """ The Path field value to override on every cookie."""
 54        self.cookie_domain = cookie_domain
 55        """ The Domain field value to override on every cookie."""
 56        self.cookie_secure = cookie_secure
 57        """ The Secure field value to override on every cookie."""
 58        self.cookie_httponly = cookie_httponly
 59        """ The HttpOnly field value to override on every cookie."""
 60        self.cookie_samesite = cookie_samesite
 61        """ The SameStime field value to override on every cookie."""
 62        self.excluded_cookies = excluded_cookies
 63        """
 64        The list of cookies to ignore when decrypting and encrypting in
 65        the request --> response cycle. This option is mutually exclusive with
 66        `included_cookies`.
 67        """
 68        self.included_cookies = included_cookies
 69        """
 70        The list of cookies to decrypt and encrypt in the request --> response cycle.
 71        This option is mutually exclusive with `excluded_cookies`.
 72        """
 73
 74        if excluded_cookies and included_cookies:
 75            raise BadArgumentError(
 76                "It doesn't make sense to specify both excluded and included cookies."
 77                " Supply with `excluded_cookies` or `included_cookies` to restrict"
 78                " which cookies should be secure."
 79            )
 80
 81        if cookie_samesite and cookie_samesite.lower() not in ["strict", "lax", "none"]:
 82            raise BadArgumentError(
 83                "SameSite attribute must be either 'strict', 'lax' or 'none'"
 84            )

 86    async def dispatch(
 87        self, request: Request, call_next: RequestResponseEndpoint
 88    ) -> Response:
 89        # request cookies are only passed along and mutable within the asgi scope
 90        headers = request.scope["headers"]
 91        raw_cookies = [h for h in headers if h[0] == b"cookie"]
 92        if len(raw_cookies):
 93            # if encrypted cookies are present, remove and parse them
 94            headers.remove(raw_cookies[0])
 95            cookies: SimpleCookie = SimpleCookie(raw_cookies[0][1].decode())
 96
 97            for cookie, morsel in list(cookies.items()):
 98                # if the cookie is included or not excluded
 99                if (
100                    (not self.included_cookies and not self.excluded_cookies)
101                    or (self.included_cookies and cookie in self.included_cookies)
102                    or (self.excluded_cookies and cookie not in self.excluded_cookies)
103                ):
104                    try:
105                        # try to decrypt the token. if you can, then it's already
106                        # encrypted and can be passed along
107                        cookies[cookie] = self.mfernet.decrypt(
108                            morsel.value.encode()
109                        ).decode()
110                    except InvalidToken:
111                        # delete invalid or unencrypted cookies if enabled
112                        if self.discard_invalid:
113                            del cookies[cookie]
114
115            # serialize and append the decrypted cookies to the asgi scope headers
116            new_cookie = (
117                b"cookie",
118                cookies.output(header="", sep=";").strip().encode(),
119            )
120            headers.append(new_cookie)
121
122        # propagate the modified request
123        response: Response = await call_next(request)
124
125        # for every cookie in the response
126        for raw_cookie in response.headers.getlist("set-cookie"):
127            # decode it
128            ncookie: SimpleCookie = SimpleCookie(raw_cookie)
129            key = [*ncookie.keys()][0]
130
131            # if the cookie is included or not excluded
132            if (
133                (not self.included_cookies and not self.excluded_cookies)
134                or (self.included_cookies and key in self.included_cookies)
135                or (self.excluded_cookies and key not in self.excluded_cookies)
136            ):
137                # create a new encrypted cookie with the desired attributes
138                response.set_cookie(
139                    key,
140                    value=self.mfernet.encrypt(ncookie[key].value.encode()).decode(),
141                    path=self.cookie_path or ncookie[key]["path"],
142                    domain=self.cookie_domain or ncookie[key]["domain"],
143                    secure=self.cookie_secure or ncookie[key]["secure"],
144                    httponly=self.cookie_httponly or ncookie[key]["httponly"],
145                    samesite=self.cookie_samesite or ncookie[key]["samesite"],
146                )
147
148        return response
